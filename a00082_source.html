171 #define _NavierStokes2DRoeAverage_(uavg,uL,uR,p) \
172  { \
173  double rho ,vx, vy, e ,P ,H ,csq, vsq; \
174  double rhoL,vxL,vyL,eL,PL,HL,cLsq,vsqL; \
175  double rhoR,vxR,vyR,eR,PR,HR,cRsq,vsqR; \
176  double gamma = p->gamma; \
177  rhoL = uL[0]; \
178  vxL = uL[1] / rhoL; \
179  vyL = uL[2] / rhoL; \
180  eL = uL[3]; \
181  vsqL = (vxL*vxL) + (vyL*vyL); \
182  PL = (eL - 0.5*rhoL*vsqL) * (gamma-1.0); \
183  cLsq = gamma * PL/rhoL; \
184  HL = 0.5*(vxL*vxL+vyL*vyL) + cLsq / (gamma-1.0); \
185  rhoR = uR[0]; \
186  vxR = uR[1] / rhoR; \
187  vyR = uR[2] / rhoR; \
188  eR = uR[3]; \
189  vsqR = (vxR*vxR) + (vyR*vyR); \
190  PR = (eR - 0.5*rhoR*vsqR) * (gamma-1.0); \
191  cRsq = gamma * PR/rhoR; \
192  HR = 0.5*(vxR*vxR+vyR*vyR) + cRsq / (gamma-1.0); \
193  double tL = sqrt(rhoL); \
194  double tR = sqrt(rhoR); \
195  rho = tL * tR; \
196  vx = (tL*vxL + tR*vxR) / (tL + tR); \
197  vy = (tL*vyL + tR*vyR) / (tL + tR); \
198  H = (tL*HL + tR*HR) / (tL + tR); \
199  vsq = vx*vx + vy*vy; \
200  csq = (gamma-1.0) * (H-0.5*vsq); \
201  P = csq * rho / gamma; \
202  e = P/(gamma-1.0) + 0.5*rho*vsq; \
203  uavg[0] = rho; \
204  uavg[1] = rho*vx; \
205  uavg[2] = rho*vy; \
206  uavg[3] = e; \
207  }
