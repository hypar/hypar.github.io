95 int MPIRanknD(int ndims,int rank,int* iproc,int *ip)
96 {
97  int i,term = 1;
98  for (i=0; i<ndims; i++) term *= iproc[i];
99  for (i=ndims-1; i>=0; i--) {
100  term /= iproc[i];
101  ip[i] = rank/term;
102  rank -= ip[i]*term;
103  }
104  return(0);
