145 
146  if (a_uex == NULL) {
147  fprintf(stderr, "Error in SparseGrids::computeError() -\n");
148  fprintf(stderr, " exact solution pointer in NULL.\n");
149  exit(1);
150  }
151 
152  /* calculate solution norms (for relative error) */
153  double sum, global_sum;
154  double solution_norm[3] = {0.0,0.0,0.0};
155  /* L1 */
156  sum = ArraySumAbsnD (solver->nvars,solver->ndims,solver->dim_local,
157  solver->ghosts,solver->index,a_uex);
158  global_sum = 0; MPISum_double(&global_sum,&sum,1,&mpi->world);
159  solution_norm[0] = global_sum/((double)solver->npoints_global);
160  /* L2 */
161  sum = ArraySumSquarenD(solver->nvars,solver->ndims,solver->dim_local,
162  solver->ghosts,solver->index,a_uex);
163  global_sum = 0; MPISum_double(&global_sum,&sum,1,&mpi->world);
164  solution_norm[1] = sqrt(global_sum/((double)solver->npoints_global));
165  /* Linf */
166  sum = ArrayMaxnD (solver->nvars,solver->ndims,solver->dim_local,
167  solver->ghosts,solver->index,a_uex);
168  global_sum = 0; MPIMax_double(&global_sum,&sum,1,&mpi->world);
169  solution_norm[2] = global_sum;
170 
171  /* compute error = difference between exact and numerical solution */
172  long size = solver->nvars*solver->npoints_local_wghosts;
173  _ArrayAXPY_(solver->u,-1.0,a_uex,size);
174 
175  /* calculate L1 norm of error */
176  sum = ArraySumAbsnD (solver->nvars,solver->ndims,solver->dim_local,
177  solver->ghosts,solver->index,a_uex);
178  global_sum = 0; MPISum_double(&global_sum,&sum,1,&mpi->world);
179  solver->error[0] = global_sum/((double)solver->npoints_global);
180 
181  /* calculate L2 norm of error */
182  sum = ArraySumSquarenD(solver->nvars,solver->ndims,solver->dim_local,
183  solver->ghosts,solver->index,a_uex);
184  global_sum = 0; MPISum_double(&global_sum,&sum,1,&mpi->world);
185  solver->error[1] = sqrt(global_sum/((double)solver->npoints_global));
186 
187  /* calculate Linf norm of error */
188  sum = ArrayMaxnD (solver->nvars,solver->ndims,solver->dim_local,
189  solver->ghosts,solver->index,a_uex);
190  global_sum = 0; MPIMax_double(&global_sum,&sum,1,&mpi->world);
191  solver->error[2] = global_sum;
