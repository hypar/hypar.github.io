36 #define _NavierStokes3DGetFlowVar_(u,rho,vx,vy,vz,e,P,p) \
37  { \
38  double gamma = p->gamma, vsq; \
39  rho = u[0]; \
40  vx = u[1] / rho; \
41  vy = u[2] / rho; \
42  vz = u[3] / rho; \
43  e = u[4]; \
44  vsq = vx*vx + vy*vy + vz*vz; \
45  P = (e - 0.5*rho*vsq) * (gamma-1.0); \
46  }

71 #define _NavierStokes3DRoeAverage_(uavg,uL,uR,p) \
72  { \
73  double rho ,vx, vy, vz, e ,P ,H; \
74  double rhoL,vxL,vyL,vzL,eL,PL,HL,cLsq; \
75  double rhoR,vxR,vyR,vzR,eR,PR,HR,cRsq; \
76  double gamma = p->gamma; \
77  _NavierStokes3DGetFlowVar_(uL,rhoL,vxL,vyL,vzL,eL,PL,p); \
78  cLsq = gamma * PL/rhoL; \
79  HL = 0.5*(vxL*vxL+vyL*vyL+vzL*vzL) + cLsq / (gamma-1.0); \
80  _NavierStokes3DGetFlowVar_(uR,rhoR,vxR,vyR,vzR,eR,PR,p); \
81  cRsq = gamma * PR/rhoR; \
82  HR = 0.5*(vxR*vxR+vyR*vyR+vzR*vzR) + cRsq / (gamma-1.0); \
83  double tL = sqrt(rhoL); \
84  double tR = sqrt(rhoR); \
85  rho = tL * tR; \
86  vx = (tL*vxL + tR*vxR) / (tL + tR); \
87  vy = (tL*vyL + tR*vyR) / (tL + tR); \
88  vz = (tL*vzL + tR*vzR) / (tL + tR); \
89  H = (tL*HL + tR*HR ) / (tL + tR); \
90  P = (H - 0.5* (vx*vx+vy*vy+vz*vz)) * (rho*(gamma-1.0))/gamma; \
91  e = P/(gamma-1.0) + 0.5*rho*(vx*vx+vy*vy+vz*vz); \
92  uavg[0] = rho; \
93  uavg[1] = rho*vx; \
94  uavg[2] = rho*vy; \
95  uavg[3] = rho*vz; \
96  uavg[4] = e; \
97  }
