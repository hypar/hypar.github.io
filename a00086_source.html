151 #define _Euler1DRoeAverage_(uavg,uL,uR,p) \
152  { \
153  double rho ,v ,e ,P ,H ,csq; \
154  double rhoL,vL,eL,PL,HL,cLsq; \
155  double rhoR,vR,eR,PR,HR,cRsq; \
156  double gamma = p->gamma; \
157  rhoL = uL[0]; \
158  vL = uL[1] / rhoL; \
159  eL = uL[2]; \
160  PL = (eL - 0.5*rhoL*vL*vL) * (gamma-1.0); \
161  cLsq = gamma * PL/rhoL; \
162  HL = 0.5*vL*vL + cLsq / (gamma-1.0); \
163  rhoR = uR[0]; \
164  vR = uR[1] / rhoR; \
165  eR = uR[2]; \
166  PR = (eR - 0.5*rhoR*vR*vR) * (gamma-1.0); \
167  cRsq = gamma * PR/rhoR; \
168  HR = 0.5*vR*vR + cRsq / (gamma-1.0); \
169  double tL = sqrt(rhoL); \
170  double tR = sqrt(rhoR); \
171  rho = tL * tR; \
172  v = (tL*vL + tR*vR) / (tL + tR); \
173  H = (tL*HL + tR*HR) / (tL + tR); \
174  csq = (gamma-1.0) * (H-0.5*v*v); \
175  P = csq * rho / gamma; \
176  e = P/(gamma-1.0) + 0.5*rho*v*v; \
177  \
178  uavg[0] = rho; \
179  uavg[1] = rho*v; \
180  uavg[2] = e; \
