84 #define _NavierStokes3DGetFlowVar_(u,rho,vx,vy,vz,e,P,p) \
85  { \

202 #define _NavierStokes3DRoeAverage_(uavg,uL,uR,p) \
203  { \
204  double rho ,vx, vy, vz, e ,P ,H; \
205  double rhoL,vxL,vyL,vzL,eL,PL,HL,cLsq; \
206  double rhoR,vxR,vyR,vzR,eR,PR,HR,cRsq; \
207  double gamma = p->gamma; \
208  _NavierStokes3DGetFlowVar_(uL,rhoL,vxL,vyL,vzL,eL,PL,p); \
209  cLsq = gamma * PL/rhoL; \
210  HL = 0.5*(vxL*vxL+vyL*vyL+vzL*vzL) + cLsq / (gamma-1.0); \
211  _NavierStokes3DGetFlowVar_(uR,rhoR,vxR,vyR,vzR,eR,PR,p); \
212  cRsq = gamma * PR/rhoR; \
213  HR = 0.5*(vxR*vxR+vyR*vyR+vzR*vzR) + cRsq / (gamma-1.0); \
214  double tL = sqrt(rhoL); \
215  double tR = sqrt(rhoR); \
216  rho = tL * tR; \
217  vx = (tL*vxL + tR*vxR) / (tL + tR); \
218  vy = (tL*vyL + tR*vyR) / (tL + tR); \
219  vz = (tL*vzL + tR*vzR) / (tL + tR); \
220  H = (tL*HL + tR*HR ) / (tL + tR); \
221  P = (H - 0.5* (vx*vx+vy*vy+vz*vz)) * (rho*(gamma-1.0))/gamma; \
222  e = P/(gamma-1.0) + 0.5*rho*(vx*vx+vy*vy+vz*vz); \
223  uavg[0] = rho; \
224  uavg[1] = rho*vx; \
225  uavg[2] = rho*vy; \
226  uavg[3] = rho*vz; \
227  uavg[4] = e; \
