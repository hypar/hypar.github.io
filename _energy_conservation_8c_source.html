86  U = (double*) calloc (dims[0]*dims[1]*nvars ,sizeof(double));
87 
88  /* read grid and solution */
89  fread(x,sizeof(double),dims[0]+dims[1] ,in);
90  fread(U,sizeof(double),dims[0]*dims[1]*nvars,in);
91  /* done reading */
92  fclose(in);

94  int imax = dims[0];
95  int jmax = dims[1];
96 
97  /* calculate primitive variables */
98  double *X = x;
99  double *Y = x+imax;
100  double grav_x = params->grav_x;
101  double grav_y = params->grav_y;
102  double gamma = params->gamma;
103 
104  int i, j;
105  *kinetic_energy = 0;
106  *potential_energy = 0;
107  *internal_energy = 0;
108  for (i=0; i<imax; i++) {
109  for (j=0; j<jmax; j++) {
110  int p = i + imax*j;
111 
112  /* cell volume */
113  double dx, dy;
114  if (i == 0) dx = X[1]-X[0];
115  else if (i == imax-1) dx = X[imax-1] - X[imax-2];
116  else dx = 0.5 * (X[i+1]-X[i-1]);
117  if (j == 0) dy = Y[1] - Y[0];
118  else if (j == jmax-1) dy = Y[jmax-1] - Y[jmax-2];
119  else dy = 0.5 * (Y[j+1]-Y[j-1]);
120  double vol = dx * dy;
121 
122  double rho, uvel, vvel, E, P, theta;
123  rho = U[nvars*p+0];
124  uvel = U[nvars*p+1] / rho;
125  vvel = U[nvars*p+2] / rho;
126  E = U[nvars*p+3];
127  P = (gamma-1.0) * (E - 0.5*rho*(uvel*uvel+vvel*vvel));
128 
129  *kinetic_energy += (0.5*rho*(uvel*uvel+vvel*vvel)*vol);
130  *potential_energy += (rho * (grav_x*X[i] + grav_y*Y[j]) * vol);
131  *internal_energy += ( P/(gamma-1.0) * vol);
132  }
133  }
134  *total_energy = *kinetic_energy + *potential_energy + *internal_energy;
135 
136  /* clean up */
137  free(U);
138  free(x);
139 }
